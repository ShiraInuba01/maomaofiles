-- 3. Block Menu Core & COMPLETE Movement Freeze
local function sinkAction()
    return Enum.ContextActionResult.Sink
end

ContextActionService:BindActionAtPriority(
    "FreezeEverything",
    sinkAction,
    false,
    3000, 
    Enum.KeyCode.Escape,
    Enum.KeyCode.ButtonStart,
    Enum.KeyCode.W, Enum.KeyCode.A, Enum.KeyCode.S, Enum.KeyCode.D,
    Enum.KeyCode.Up, Enum.KeyCode.Down, Enum.KeyCode.Left, Enum.KeyCode.Right,
    Enum.KeyCode.Space
)

local freezeLoop = RunService.RenderStepped:Connect(function()
    if GuiService.MenuIsOpen then
        GuiService:CloseMenuWindow()
    end
    
    local character = player.Character
    if character then
        local rootPart = character:FindFirstChild("HumanoidRootPart")
        local humanoid = character:FindFirstChildOfClass("Humanoid")
        if rootPart then
            rootPart.Velocity = Vector3.zero
            rootPart.RotVelocity = Vector3.zero
        end
        if humanoid then
            humanoid.WalkSpeed = 0
            humanoid.JumpPower = 0
        end
    end
end)

-- 4. Infinite Animation Logic
local function startLoading()
	local looping = true
	
	-- Text bounce / wacky jitter loop
	task.spawn(function()
		while looping do
			local t1 = TweenService:Create(loadingText, TweenInfo.new(0.6, Enum.EasingStyle.Back, Enum.EasingDirection.Out), {TextSize = 76})
			t1:Play()
			task.wait(0.6)
			local t2 = TweenService:Create(loadingText, TweenInfo.new(0.6, Enum.EasingStyle.Back, Enum.EasingDirection.Out), {TextSize = 72})
			t2:Play()
			task.wait(0.6)
		end
	end)

	-- ENDLESS MOVING LIQUID RED/BLUE WAVE
	task.spawn(function()
		while looping do
			gradient.Offset = Vector2.new(-1, 0)
			local liquidTween = TweenService:Create(gradient, TweenInfo.new(1.8, Enum.EasingStyle.Linear), {Offset = Vector2.new(1, 0)})
			liquidTween:Play()
			liquidTween.Completed:Wait()
		end
	end)

	-- Smoothly fill up to 95% and stay stuck
	local tweenInfo = TweenInfo.new(5, Enum.EasingStyle.Quart, Enum.EasingDirection.Out)
	local fillTween = TweenService:Create(barFill, tweenInfo, {Size = UDim2.new(0.95, 0, 1, 0)})
	fillTween:Play()
end

task.spawn(startLoading)
